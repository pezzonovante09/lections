ls - показывает все файлы и папки внутри текущей директории
ls -a - показывает все скрытые файлы и папки
ls -l - показывает более подробную информацию о файлах и папках

ДИРЕКТОРИЯ == ПАПКА

cd - переходит в домашнюю папку
cd staryi_bog - переходит в директорию staryi_bog
cd .. - вовзращается на одну директорию назад
cd ../.. - вовзращает на 3 директории назад
mkdir test - создаст папку с название s в текущей директории
mkdir test/test2 - создаст папку с названием test2 в папке с названием test в текущей директории test
(если test существует, в ином случае кинет ошибку)
mkdir test1 test2 test 3 - создает 3 директории разом, в текущей папке

touch text1.txt - создает файл test1.txt в текущей директории
touch text1.py text2.py - создает несколько файлов разом, в текущей директории

rm test.py - удаляет файл test.py
rmdir - удаляет только пустую папку(файлы не удаляте)
rm -rf - test (test1.txt) - удаляет и папки и файлы принуждено, не смотря на наличие или отсутствие 
файлов/папок, внутри удаляемой директории


mv test.txt test/ - переносит файл из test.txt в директорию test
mv test.txt test2.txt - переименовывает файл test.txt в test2.txt
mv test1/ inner-lection/ - переименовывает папку test1 в inner-lection

history - показывает все команды, которые были выполнены в данном терминале
history 10 - показывает последние 10 команд в данном терминале
history | grep cd - показывает все команды с использованиием слова cd

pwd - (print working diractory) - показывает абслютный(начиная с корневой папки/)
путь до текущей директории 

cat test.txt - выводит все содержимое файла test.txt в терминал
nano test.py - открывает файл test.py в редакторе кода nano
code test.py - открывает файл test.py в редакторе vs code
code . - открывает vs code в текущей директории
